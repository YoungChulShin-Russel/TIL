    println(describe(2))	// Not a string
    println(describe("kkkk"))	// Unkown
}

// any를 받아서 String을 리턴
// any는 kotlin에서 object
// Expression 
fun describe(obj: Any) : String = 
    when (obj) {
        1 			-> "One"
        "Hello" 	-> "Greeting"
        is Long 	-> "Long"
        !is String	-> "Not a string"
        else		-> "Unknown"
    }
